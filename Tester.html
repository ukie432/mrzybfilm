// 1. Variabel global untuk menampung data film dari JSON let allMovies = []; // 2. Fungsi untuk mengambil data dari file Tester.json di GitHub async function loadMoviesFromJSON() { const url = 'data/Tester.json'; try { const response = await fetch(url); if (!response.ok) throw new Error('Gagal memuat data JSON'); allMovies = await response.json(); console.log('Data film berhasil dimuat:', allMovies); // Opsional: Tampilkan semua film di halaman awal setelah data dimuat // renderMovies(allMovies); } catch (error) { console.error('Error saat mengambil data:', error); } } // Panggil fungsi pemuatan data saat halaman web selesai dimuat window.addEventListener('DOMContentLoaded', loadMoviesFromJSON); // 3. Logika Fungsi Pencarian yang Diperbaiki const searchInput = document.getElementById('search-input'); searchInput.addEventListener('input', (e) => { const searchTerm = e.target.value.trim().toLowerCase(); // Jika kolom pencarian kosong, tampilkan kembali semua film/kembalikan ke awal if (searchTerm === '') { // Panggil fungsi untuk menampilkan semua daftar film semula renderMovies(allMovies); return; } // Lakukan filter pencarian berdasarkan judul (title) atau deskripsi (description) const filteredMovies = allMovies.filter(movie => { const titleMatch = movie.title && movie.title.toLowerCase().includes(searchTerm); const descMatch = movie.description && movie.description.toLowerCase().includes(searchTerm); return titleMatch || descMatch; }); // Tampilkan hasil filter ke container HTML if (filteredMovies.length > 0) { renderMovies(filteredMovies); } else { // Jika tidak ada yang cocok, tampilkan pesan error/kosong const container = document.getElementById('popular-movies'); // sesuaikan ID container Anda container.innerHTML = `
Film "${e.target.value}" tidak ditemukan.
`; } }); // 4. Fungsi Pembantu untuk Merender/Memasukkan Kartu Film ke HTML secara Dinamis function renderMovies(moviesList) { const container = document.getElementById('popular-movies'); // Sesuaikan dengan ID container di HTML Anda container.innerHTML = ''; // Kosongkan isi kontainer lama moviesList.forEach(movie => { // Pastikan properti seperti movie.title, movie.poster, movie.year sesuai dengan struktur properti di Tester.json Anda const movieCard = document.createElement('div'); movieCard.className = 'movie-card'; movieCard.onclick = () => openModal(movie.title); // Sesuaikan dengan fungsi modal Anda movieCard.innerHTML = `

${movie.title}

${movie.year || ''} ${movie.rating || '0.0'}

${movie.description || ''}

`; container.appendChild(movieCard); }); }
